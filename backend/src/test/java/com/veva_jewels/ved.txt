}private byte[] readFrame(InputStream in) throws IOException {

    if (isIRS) {
        ByteArrayOutputStream buffer = new ByteArrayOutputStream();

        int b;
        while ((b = in.read()) != -1) {

            if (b == '\n') { // end of ASCII line
                break;
            }

            buffer.write(b);

            if (buffer.size() > 1024) break;
        }

        if (buffer.size() == 0) return null;

        return buffer.toByteArray();
    }

    // 👉 existing binary logic
    ByteArrayOutputStream buffer = new ByteArrayOutputStream();
    int b;
    int count = 0;

    while ((b = in.read()) != -1) {
        if (b == 0x5A) {
            buffer.write(b);
            break;
        }
        if (++count > 1024) return null;
    }

    if (b == -1) return null;

    while ((b = in.read()) != -1) {
        buffer.write(b);
        if (b == 0xA5) break;
        if (buffer.size() > 2048) return null;
    }

    return buffer.toByteArray();
}

private HardwareData parseIRS(String response) {

    HardwareData data = new HardwareData();

    try {
        String[] parts = response.split(",");

        for (String part : parts) {

            if (part.contains("VSWR")) {
                data.vswr = Double.parseDouble(part.split("=")[1]);
            } else if (part.contains("TEMP")) {
                data.temp = Integer.parseInt(part.split("=")[1]);
            } else if (part.contains("FWD")) {
                data.fwd = Integer.parseInt(part.split("=")[1]);
            } else if (part.contains("RFL")) {
                data.rfl = Integer.parseInt(part.split("=")[1]);
            }
        }

        data.paOn = true;

    } catch (Exception e) {
        System.out.println("IRS Parse Error: " + response);
    }

    return data;
}

private boolean isSpecialBand(String name) {
    String band = name.replaceAll("\\s", "").toLowerCase();
    return band.equals("2400-2500mhz") ||
           band.equals("136-174mhz") ||
           band.equals("5700-5900mhz");
}
