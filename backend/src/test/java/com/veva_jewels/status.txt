private void handleFrame(byte[] frame) {

    // 🔥 ASCII handling
    if (isAsciiBased) {
        handleAsciiResponse(frame);
        return;
    }

    // 🔥 Normal binary handling
    if (!verifyXor(frame)) {
        System.out.println("XOR failed: " + bytesToHex(frame));
        return;
    }

    HardwareData data = parseFrame(frame);
    this.lastData = data;

    if (dataCallback != null) {
        dataCallback.accept(data);
    }

    System.out.println("VSWR: " + data.vswr +
            " TEMP: " + data.temp +
            " FWD: " + data.fwd +
            " RFL: " + data.rfl);
}


private void handleAsciiResponse(byte[] frame) {

    String response = new String(frame).trim();
    System.out.println("ASCII Response: " + response);

    HardwareData data = new HardwareData();

    // 🔥 Basic ON/OFF detection (adjust if needed)
    if (response.contains("1") || response.toUpperCase().contains("ON")) {
        data.paOn = true;
    } else {
        data.paOn = false;
    }

    // No telemetry available
    data.vswr = 0;
    data.temp = 0;
    data.fwd = data.paOn ? 1 : 0;
    data.rfl = 0;

    this.lastData = data;

    if (dataCallback != null) {
        dataCallback.accept(data);
    }
}


private void startAutoPolling() {

    // 🔥 skip polling for ASCII bands
    if (isAsciiBased) {
        return;
    }

    if (statusCommand == null) return;

    scheduler = Executors.newSingleThreadScheduledExecutor();

    scheduler.scheduleAtFixedRate(() -> {
        if (connected.get() && statusCommand != null) {
            enqueueCommand(statusCommand);
        }
    }, 0, 1, TimeUnit.SECONDS);
}


private boolean isAsciiFrame(byte[] frame) {
    for (byte b : frame) {
        if (b < 0x20 || b > 0x7E) {
            return false;
        }
    }
    return true;
}
